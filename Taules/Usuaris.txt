CREATE TABLE `usuaris_botiga` (
  `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `usuari` VARCHAR(45) NOT NULL DEFAULT '',
  `password` VARCHAR(45) NOT NULL DEFAULT '',
  `nom` VARCHAR(45) NOT NULL DEFAULT '',
  `cognoms` VARCHAR(60) NOT NULL DEFAULT '',
  `email` VARCHAR(60) NOT NULL DEFAULT '',
  `direccio` VARCHAR(80) NOT NULL DEFAULT '',
  `poblacio` VARCHAR(45) NOT NULL DEFAULT '',
  `codiPostal` VARCHAR(45) NOT NULL DEFAULT '',
  `rol` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  PRIMARY KEY(`id`)
)
