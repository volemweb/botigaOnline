CREATE TABLE `botigaonline`.`comandes` (
  `Id_Comanda` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `Id_Usuari` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  `data` DATE NOT NULL DEFAULT 0,
  `num_productes` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  `total` DOUBLE NOT NULL DEFAULT 0,
  `estat` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  PRIMARY KEY(`Id_comanda`)
)
ENGINE = InnoDB;
