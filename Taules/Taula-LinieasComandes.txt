CREATE TABLE `botigaonline`.`lineas_comandes` (
  `Id_linea` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `Id_comanda` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  `producte` VARCHAR(45) NOT NULL DEFAULT '',
  `quantitat` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  `preu` DOUBLE NOT NULL DEFAULT 0,
  PRIMARY KEY(`Id_linea`)
)
ENGINE = InnoDB;
